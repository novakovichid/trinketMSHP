from js import document
from pyodide.ffi import create_proxy
import sys
import traceback

console = document.getElementById("console-output")
status = document.getElementById("runtime-status")
textarea = document.getElementById("pyscript-code")


def append(text):
    console.textContent = console.textContent + text


class ConsoleWriter:
    def write(self, data):
        if data:
            append(data)

    def flush(self):
        return None


sys.stdout = ConsoleWriter()
sys.stderr = ConsoleWriter()


def run_code(event=None):
    console.textContent = ""
    status.textContent = "Выполнение..."
    try:
        exec(textarea.value, {})
    except Exception:
        traceback.print_exc()
    status.textContent = "Готово"


document.getElementById("run").addEventListener("click", create_proxy(run_code))
